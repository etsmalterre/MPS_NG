// Export entreprise table from HFSQL to CSV
// Target: C:\dev\MPS_NG\data_migration\data\entreprise.csv
//
// Usage: Call this procedure from a WinDev project connected to the MPS HFSQL database.
// The CSV output uses UTF-8 encoding with headers matching the PostgreSQL schema.

PROCEDURE ExportEntreprise()

// Internal procedure: escape a field value for CSV output
PROCEDURE INTERNE CsvEscapeField(LOCAL sValue is string)
	IF sValue = "" THEN
		RENVOYER ""
	END
	sResult is string = Replace(sValue, """", """""")
	IF Contains(sResult, ",") OR Contains(sResult, """") OR Contains(sResult, CR) OR Contains(sResult, Charact(10)) THEN
		sResult = """" + sResult + """"
	END
	RENVOYER sResult
FIN

sOutputPath is string = "C:\dev\MPS_NG\data_migration\data\entreprise.csv"

// Build CSV content in memory
sCsv is string = "nom,commentaire"

// Read all records from entreprise
HReadFirst(entreprise, IDentreprise)
WHILE NOT HOut()
	sNom is string = CsvEscapeField(entreprise.nom)
	sCommentaire is string = CsvEscapeField(entreprise.commentaire)

	sCsv += CR + sNom + "," + sCommentaire

	HReadNext(entreprise, IDentreprise)
END

// Save as UTF-8
fSaveText(sOutputPath, StringToUTF8(sCsv))

Trace("Export done: " + sOutputPath)
RETURN True
